#Definimos la matriz de correlacin como la fuente de nuestros datos
corr = df.corr()

#Definimos el tamao de la figura para asegurar que se observen los valores interiores
ax = plt.subplots(figsize=(24,18))

#Generamos una mscara para evitar tener un espejo en la parte superior de la grfica y simplificar la observacin
mask = np.zeros_like(corr)
mask[np.triu_indices_from(mask)] = True

#Usando seaborn heatmap como tipo de grfica, colocamos los valores, solicitamos que incluya las anotaciones y seleccionamos un mapa de color,
# en este caso utilizamos "viridis" y seleccionamos la mscara para no mostrar los datos duplicados, iniciando por la diagonal de unos en la 
# parte central de la grfica
ax = sbn.heatmap(corr,annot=True,mask = mask) # cmap='viridis'
